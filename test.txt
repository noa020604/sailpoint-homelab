# --- Stage 1: Build dependencies ---
FROM python:3.11-slim AS builder

WORKDIR /app

# התקנת כלי בנייה במידה ויהיה צורך בעתיד (במקרה של Flask פשוט אין צורך, אבל זה טוב לגרסה גנרית)
RUN apt-get update && apt-get install -y --no-install-recommends build-essential && rm -rf /var/lib/apt/lists/*

COPY requirements.txt .

# התקנת ה-Dependencies לתיקייה ייעודית (wheels) כדי שנוכל להעתיק אותן בקלות לשלב הבא
RUN pip install --no-cache-dir --user -r requirements.txt


# --- Stage 2: Final lightweight runtime ---
FROM python:3.11-slim AS runner

WORKDIR /app

# יצירת משתמש מערכת לא-פריווילגי (לא מריצים קונטיינרים כ-root בפרודקשן!)
RUN groupadd -g 10001 appuser && \
    useradd -u 10001 -g appuser -s /sbin/nologin -c "App User" appuser

# העתקת ה-dependencies שהותקנו בשלב הקודם מה-Home של ה-builder
COPY --from=builder /root/.local /home/appuser/.local
COPY app.py .

# הגדרת משתני סביבה חיוניים ל-Python ול-Path של המשתמש החדש
ENV PATH=/home/appuser/.local/bin:$PATH
ENV PYTHONUNBUFFERED=1

# שינוי בעלות על התיקייה למשתמש המאובטח
RUN chown -R appuser:appuser /app /home/appuser

# מעבר למשתמש הלא-פריווילגי
USER appuser

EXPOSE 8080

CMD ["python", "app.py"]
